 98     //Bubble Sort switches elements in an array (one after the next) 
 99     //Time Complexity: O(n^2)
100     static int[] BubbleSort(int[] arr){
101         boolean isSorted = false;
102         int temp;
103         while(!isSorted){
104             isSorted = true;
105             for(int i = 0; i < arr.length - 1; i ++){
106                 if(arr[i] > arr[i+1]){
107                     isSorted = false;
108                     temp = arr[i];
109                     arr[i] = arr[i+1];
110                     arr[i+1] = temp;
111                 }
112             }
113         }
114         return arr;
115     }

124     static int[] MergeSort(int[] arr, int len){
125         int[] mirrorArr = arr.clone();
126         int a = 0,b = len, indx = 0, upperA, upperB;
127         while(a <= arr.length - 1 && b <= arr.length - 1){
128             upperA = b;
129             upperB = (b + len < arr.length) ? b + len : arr.length;
130             while((a < upperA) || (b < upperB)){ 
131                 if(a == upperA){
132                     arr[indx++] = mirrorArr[b++];
133                 }else{
134                     if((b == upperB) || (b >= mirrorArr.length)){
135                         arr[indx++] = mirrorArr[a++];
136                     }else{
137                         if(mirrorArr[a] < mirrorArr[b]){
138                             arr[indx++] = mirrorArr[a++];
139                         }else{
140                             arr[indx++] = mirrorArr[b++];
141                         }
142                     }
143                 }
144             }
145             a+= len;
146             b+= len;
147         }
148         return (len > arr.length) ? arr : MergeSort(arr, len*2);
149     }

151     // A utility function to get maximum value in arr[] 
152     static int getMax(int array[], int n) 
153     { 
154         int mx = arr[0]; 
155         for (int i = 1; i < n; i++) 
156             if (array[i] > mx) 
157                 mx = array[i]; 
158         return mx; 
159     }

161     // A function to do counting sort of arr[] according to 
162     // the digit represented by exp. 
163     static void countSort(int array[], int n, int exp) 
164     { 
165         int output[] = new int[n]; // output array 
166         int i; 
167         int count[] = new int[10]; 
168         Arrays.fill(count,0); 
169   
170         // Store count of occurrences in count[] 
171         for (i = 0; i < n; i++) 
172             count[ (array[i]/exp)%10 ]++; 
173   
174         // Change count[i] so that count[i] now contains actual position of this digit in output[] 
175         for (i = 1; i < 10; i++){ 
176             count[i] += count[i - 1]; 
177         }
178         
179         // Build the output array 
180         for (i = n - 1; i >= 0; i--) { 
181             output[count[ (array[i]/exp)%10 ] - 1] = array[i]; 
182             count[ (array[i]/exp)%10 ]--; 
183         } 
184   
185         // Copy the output array to arr[], so that arr[] now 
186         // contains sorted numbers according to curent digit 
187         for (i = 0; i < n; i++) 
188             array[i] = output[i]; 
189     }

191     // Radix Sort 
192     static void radixSort(int array[], int n) 
193     { 
194         // Find the maximum number to know number of digits 
195         int m = getMax(array, n);         
196         // Do counting sort for every digit. Instead of passing digit number, exp is passed. exp is 10^i where i is current digit number 
197         for (int exp = 1; m/exp > 0; exp *= 10) 
198             countSort(array, n, exp); 
199     }
